        name: config.name,
        type: config.type,
        
        // 尺寸
        rows: config.rows,
        cols: config.cols,
        
        // ⭐ 新增：Grid定位
        gridPosition: {
            column: '7 / span 8',  // CSS Grid column
            row: '1'               // CSS Grid row
        },
        
        // ⭐ 新增：编号配置
        numberingMode: config.numberingMode,  // 编号方式
        startNumber: config.startNumber,
        
        // ⭐ 新增：样式配置
        styles: { backgroundColor, borderColor, ... },
        
        // 数据
        seating: []
    }
}

## 🔧 核心函数

### 初始化布局

```javascript
function initializeMultiBlockLayout(layoutName) {
    const config = multiBlockLayoutConfigs[layoutName]
    seatingAreas.value = []
    
    // 遍历配置并创建座位区块
    Object.entries(config.areas).forEach(([zoneName, zoneConfig]) => {
        zoneConfig.regions.forEach((areaConfig, idx) => {
            const area = createSeatingArea({
                ...areaConfig,
                zone: zoneName,
                order: idx
            })
            
            // 生成座位
            generateSeats(area)
            
            seatingAreas.value.push(area)
        })
    })
}
```

### 获取Grid定位

```javascript
function getGridPositionStyle(area) {
    return {
        gridColumn: area.gridPosition.column,
        gridRow: area.gridPosition.row
    }
}
```

### 生成座位

```javascript
function generateSeats(area) {
    const { rows, cols, numberingMode, startNumber } = area
    
    for (let row = 0; row < rows; row++) {
        for (let col = 0; col < cols; col++) {
            const seatNumber = SeatNumberGenerator.generateSeatNumber({
                mode: numberingMode,
                row, col, rows, cols,
                baseNum: startNumber
            })
            
            area.seating.push({
                number: seatNumber,
                row, col,
                status: 'empty',
                person: null
            })
        }
    }
}
```

## 🤔 常见疑问

### Q: 如何支持更多区块？
**A**: 只需在配置中添加更多区块对象，系统会自动生成。无需修改代码逻辑。

### Q: 如何改变布局（从会场变剧院）？
**A**: 调用 `initializeMultiBlockLayout('theaterLayout')` 即可切换到预定义的剧院布局。

### Q: 响应式设计怎么做？
**A**: 本指南包含了5个断点的完整CSS Media Query，直接复制即可。

### Q: 座位编号能定制吗？
**A**: 可以。每个区块有 `numberingMode` 和 `startNumber` 两个参数可定制。支持5种编号方式。

### Q: 能否添加人员分配功能？
**A**: 可以。座位对象预留了 `person` 字段，可以存储分配的人员信息。
