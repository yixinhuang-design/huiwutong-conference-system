### 2.4 Step 4：实现核心函数

#### 2.4.1 Vue 3 Setup 函数

```javascript
import { ref, reactive, computed } from 'vue'
import SeatNumberGenerator from './seat-number-generator.js'

export default {
    setup() {
        // ========== 状态定义 ==========
        
        // 所有布局配置
        const layoutConfigs = reactive({
            conferenceHall: {
                // ... 见 2.1.1 的完整配置
            }
        })
        
        // 当前选中的布局名称
        const currentLayout = ref('conferenceHall')
        
        // 所有座位区块
        const seatingAreas = ref([])
        
        // 当前选中的座位
        const selectedSeats = reactive({})
        
        // ========== 关键函数 ==========
        
        /**
         * 初始化多块布局
         * @param {string} layoutName - 布局配置名称
         */
        function initializeMultiBlockLayout(layoutName) {
            const layoutConfig = layoutConfigs[layoutName]
            if (!layoutConfig) {
                console.error(`Layout '${layoutName}' not found`)
                return
            }
            
            seatingAreas.value = []
            currentLayout.value = layoutName
            
            // 遍历所有区域（top, bottom, center 等）
            Object.entries(layoutConfig.areas).forEach(([zoneName, zoneConfig]) => {
                // 遍历区域内的所有座位区块
                zoneConfig.regions.forEach((areaConfig, index) => {
                    // 创建座位区块
                    const area = {
                        // 基础信息
                        id: areaConfig.id,
                        name: areaConfig.name,
                        type: areaConfig.type,
                        description: areaConfig.description || '',
                        
                        // 尺寸
                        rows: areaConfig.rows,
                        cols: areaConfig.cols,
                        
                        // 编号配置
                        numberingMode: areaConfig.numberingMode,
                        startNumber: areaConfig.startNumber,
                        
                        // 布局定位
                        gridPosition: areaConfig.gridPosition,
                        zone: zoneName,
                        order: index,
                        
                        // 样式
                        styles: areaConfig.styles || {},
                        
                        // 功能标志
                        isSelectable: areaConfig.isSelectable !== false,
                        isEditable: areaConfig.isEditable !== false,
                        canAssignPeople: areaConfig.canAssignPeople !== false,
                        allowEmptySeat: areaConfig.allowEmptySeat !== true,
                        
                        // 座位数据（待生成）
                        seating: []
                    }
                    
                    // 生成座位
                    generateSeats(area)
                    
                    // 添加到列表
                    seatingAreas.value.push(area)
                })
            })
            
            console.log(`✅ Initialized layout '${layoutName}' with ${seatingAreas.value.length} areas`)
        }
        
        /**
         * 为座位区块生成座位
         * @param {Object} area - 座位区块对象
         */
        function generateSeats(area) {
            const { rows, cols, numberingMode, startNumber } = area
            area.seating = []
            
            for (let row = 0; row < rows; row++) {
                for (let col = 0; col < cols; col++) {
                    // 使用座位编号生成器
                    const seatNumber = SeatNumberGenerator.generateSeatNumber({
                        mode: numberingMode,
                        row,
                        col,
                        rows,
                        cols,
                        baseNum: startNumber
                    })
                    
                    // 创建座位对象
                    area.seating.push({
                        number: seatNumber,
                        row,
                        col,
                        status: 'empty',  // empty | occupied | selected | disabled
                        person: null,
                        assigned: false
                    })
                }
            }
        }
        
        /**
         * 获取Grid定位样式
         * @param {Object} area - 座位区块
         * @returns {Object} CSS样式对象
         */
        function getGridPositionStyle(area) {
            const { column, row } = area.gridPosition
            return {
                gridColumn: column,
                gridRow: row
            }
        }
        
        /**
         * 切换布局
         * @param {string} layoutName - 新布局名称
         */
        function switchLayout(layoutName) {
            initializeMultiBlockLayout(layoutName)
        }
        
        /**
         * 选择座位
         * @param {Object} area - 座位区块
         * @param {Object} seat - 座位对象
         */
        function selectSeat(area, seat) {
            if (!area.isSelectable || seat.status !== 'empty') {
                return
            }
            
            const seatKey = `${area.id}-${seat.number}`
            
            if (seat.status === 'selected') {
                // 取消选择
                seat.status = 'empty'
                delete selectedSeats[seatKey]
            } else {
                // 选择座位
                seat.status = 'selected'
                selectedSeats[seatKey] = {
                    areaId: area.id,
                    areaName: area.name,
                    seatNumber: seat.number
                }
            }
        }
        
        /**
         * 分配人员到座位
         * @param {Object} area - 座位区块
         * @param {Object} seat - 座位
         * @param {Object} person - 人员信息
         */
        function assignPersonToSeat(area, seat, person) {
            if (!area.canAssignPeople || seat.status !== 'empty') {
                return false
            }
            
            seat.person = person
            seat.status = 'occupied'
            seat.assigned = true
            
            return true
        }
        
        /**
         * 获取区域内已占用的座位数
         * @param {Object} area - 座位区块
         * @returns {number} 已占用座位数
         */
        function getOccupiedCount(area) {
            return area.seating.filter(s => s.status === 'occupied').length
        }
        
        /**
         * 获取所有已选择的座位
         * @returns {Array} 已选座位列表
         */
        function getSelectedSeats() {
            return Object.values(selectedSeats)
        }
        
        /**
         * 获取座位区块的总座位数
         * @param {Object} area - 座位区块
         * @returns {number} 总座位数
         */
        function getTotalSeats(area) {
            return area.seating.length
        }
        
        /**
         * 清除所有选择
         */
        function clearSelection() {
            seatingAreas.value.forEach(area => {
                area.seating.forEach(seat => {
                    if (seat.status === 'selected') {
                        seat.status = 'empty'
                    }
                })
            })
            Object.keys(selectedSeats).forEach(key => {
                delete selectedSeats[key]
            })
        }
        
        // 初始化默认布局
        initializeMultiBlockLayout('conferenceHall')
        
        // ========== 返回公开API ==========
        return {
            // 状态
            layoutConfigs,
            currentLayout,
            seatingAreas,
            selectedSeats,
            
            // 函数
            initializeMultiBlockLayout,
            switchLayout,
            selectSeat,
            assignPersonToSeat,
            getGridPositionStyle,
            getOccupiedCount,
            getSelectedSeats,
            getTotalSeats,
            clearSelection
        }
    }
}
```

## 6. 故障排除

### 问题1：座位区块位置不正确

**症状**: 座位区块显示在错误的位置

**原因**: gridPosition 定义错误

**解决方案**:
```javascript
// ❌ 错误
gridPosition: { column: 7, row: 1 }  // 应该是字符串

// ✅ 正确
gridPosition: { column: '7 / span 8', row: '1' }
```

### 问题2：座位编号重复

**症状**: 多个座位有相同的编号

**原因**: startNumber 重复或编号模式不匹配

**解决方案**:
```javascript
// 确保每个区块的 startNumber 独立
{
    id: 'area1',
    startNumber: 1,    // 1-50
    rows: 5, cols: 10
},
{
    id: 'area2',
    startNumber: 101,  // 101-150（不与area1重复）
    rows: 5, cols: 10
}
```

### 问题3：响应式布局错位

**症状**: 在某些屏幕尺寸下布局混乱

**原因**: Media Query 断点设置不合理

**解决方案**:
```css
/* 使用 min-width 而不是混合 */
@media (max-width: 1024px) { /* ✅ */ }
@media screen and (max-width: 1024px) { /* ✅ */ }

/* 避免重叠的断点 */
@media (max-width: 1023px) { /* ✅ */ }
@media (max-width: 1024px) { /* ❌ 重叠 */ }
```

## 7. 性能优化

### 7.1 优化技巧

| 优化项 | 说明 | 收益 |
|--------|------|------|
| 虚拟滚动 | 大量座位时只渲染可见区域 | 减少DOM节点 |
| 防抖选择 | 合并多次点击事件 | 减少重排 |
| CSS Grid 原生 | 充分利用GPU加速 | 性能提升 30% |
| 懒加载 | 延迟加载座位数据 | 首屏加载快 |

### 7.2 优化代码示例

```javascript
// 虚拟滚动实现
import { VirtualScroller } from '@vueuse/components'

// 防抖选择
import { debounce } from 'lodash-es'
const selectSeatDebounced = debounce(selectSeat, 200)

// 使用 computed 缓存计算结果
const totalSeats = computed(() => {
    return seatingAreas.value.reduce((sum, area) => sum + area.seating.length, 0)
})

// 批量更新座位状态
function batchUpdateSeats(updates) {
    // 一次性更新，避免多次重排
    updates.forEach(({ areaId, seatNumber, status }) => {
        const area = seatingAreas.value.find(a => a.id === areaId)
        if (area) {
            const seat = area.seating.find(s => s.number === seatNumber)
            if (seat) seat.status = status
        }
    })
}
```
