## 🔧 前端对接要点

### 数据初始化修改

**修改前**：
```javascript
data() {
    return {
        // 从LocalStorage加载
        const savedTasks = localStorage.getItem('tasks_' + conferenceId);
        if (savedTasks) {
            this.tasks = JSON.parse(savedTasks);
        }
    }
}
```

**修改后**：
```javascript
data() {
    return {
        // 从后端加载任务
        this.loadTasks();
    }
},
methods: {
    async loadTasks() {
        try {
            const data = await TaskAPI.getTasks(this.conference.id);
            this.tasks = data.records || [];
            this.updateStats();
        } catch (error) {
            console.error('加载任务失败:', error);
            this.$showToast('加载任务失败', 'error');
        }
    },
    
    updateStats() {
        this.taskStats.total = this.tasks.length;
        this.taskStats.completed = this.tasks.filter(t => t.status === 'completed').length;
        this.taskStats.inProgress = this.tasks.filter(t => t.status === 'in_progress').length;
    }
}
```

### 创建任务对接

**修改前**：
```javascript
async saveTask() {
    // 保存到LocalStorage
    localStorage.setItem('tasks_' + this.conference.id, JSON.stringify(this.tasks));
}
```

**修改后**：
```javascript
async saveTask() {
    try {
        // 调用后端API创建/更新任务
        if (this.isEditMode) {
            await TaskAPI.updateTask(this.currentTask.id, this.currentTask);
        } else {
            const result = await TaskAPI.createTask(this.currentTask);
            this.currentTask.id = result.id;
        }
        
        // 重新加载任务列表
        await this.loadTasks();
        
        this.$showToast('任务保存成功', 'success');
        this.showCreateForm = false;
    } catch (error) {
        console.error('保存任务失败:', error);
        this.$showToast('保存失败：' + error.message, 'error');
    }
}
```

### 任务列表对接

**修改前**：
```javascript
computed: {
    filteredTasks() {
        // 从本地数据筛选
        return this.tasks.filter(...)
    }
}
```

**修改后**：
```javascript
computed: {
    filteredTasks() {
        // 从后端数据筛选
        let result = this.tasks;
        if (this.taskFilter !== 'all') {
            result = result.filter(t => t.status === this.taskFilter);
        }
        return result;
    }
},
async mounted() {
    // 从后端加载
    await this.loadTasks();
}
```
