```java
@Autowired
private RegistrationAuditLogService auditLogService;

@PostMapping("/audit")
public Result<String> auditRegistration(
        @Valid @RequestBody RegistrationAuditRequest request,
        @RequestHeader(value = "X-User-Id", required = false) Long auditorId) {
    if (auditorId == null) {
        auditorId = TenantContextHolder.getUserId();
    }
    
    // 执行审核
    registrationService.auditRegistration(request, auditorId != null ? auditorId : 0L);
    
    // 记录审核日志
    Registration registration = registrationService.getById(request.getRegistrationId());
    String auditorName = "管理员"; // 可从用户服务获取
    auditLogService.logAudit(
        request.getRegistrationId(),
        auditorId,
        auditorName,
        request.getApproved() != null && request.getApproved() ? "approve" : "reject",
        request.getComment()
    );
    
    return Result.ok("审核成功");
}
```

```javascript
const saveGroupDetail = async () => {
    try {
        // 验证
        if (!currentGroup.value.name) {
            this.$showToast('请输入分组名称', 'warning');
            return;
        }
        
        // 调用后端API保存
        if (currentGroup.value.id) {
            // 更新现有分组
            await GroupAPI.updateGroup(currentGroup.value.id, {
                name: currentGroup.value.name,
                description: currentGroup.value.description,
                color: currentGroup.value.color,
                icon: currentGroup.value.icon,
                sort: currentGroup.value.sort
            });
        } else {
            // 创建新分组
            const result = await GroupAPI.createGroup({
                conferenceId: conference.value.id,
                name: currentGroup.value.name,
                description: currentGroup.value.description,
                color: currentGroup.value.color,
                icon: currentGroup.value.icon,
                type: 'manual',
                sort: currentGroup.value.sort
            });
            currentGroup.value.id = result.id;
        }

        // 更新前端状态
        const index = groups.value.findIndex(g => g.id === currentGroup.value.id);
        if (index > -1) {
            groups.value[index] = { ...currentGroup.value };
        } else {
            groups.value.push({ ...currentGroup.value });
        }

        showGroupDetailModal.value = false;
        this.$showToast('分组保存成功', 'success');
    } catch (error) {
        console.error('保存分组失败:', error);
        this.$showToast('保存失败：' + error.message, 'error');
    }
};
```
