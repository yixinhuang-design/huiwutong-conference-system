## 十三、自动化测试脚本 (Bash)

以下脚本可一键执行所有测试:

```bash
#!/bin/bash
# file: run_all_tests.sh
# 使用方法: bash run_all_tests.sh <TOKEN>

TOKEN=$1
TENANT_ID=2027317834622709762
MEETING_ID=2030309010523144194

PASS=0
FAIL=0

run_test() {
    local name=$1
    local method=$2
    local url=$3
    local data=$4
    local expected_code=${5:-200}
    
    if [ "$method" == "GET" ]; then
        RESP=$(curl -s -w "\n%{http_code}" -X GET "$url" \
          -H "Authorization: Bearer ${TOKEN}" \
          -H "X-Tenant-Id: ${TENANT_ID}")
    else
        RESP=$(curl -s -w "\n%{http_code}" -X $method "$url" \
          -H "Authorization: Bearer ${TOKEN}" \
          -H "X-Tenant-Id: ${TENANT_ID}" \
          -H "Content-Type: application/json" \
          -d "$data")
    fi
    
    HTTP_CODE=$(echo "$RESP" | tail -1)
    BODY=$(echo "$RESP" | sed '$d')
    API_CODE=$(echo "$BODY" | jq -r '.code // empty' 2>/dev/null)
    
    if [ "$API_CODE" == "$expected_code" ]; then
        echo "  ✅ PASS: $name (HTTP:$HTTP_CODE, API:$API_CODE)"
        PASS=$((PASS+1))
    else
        echo "  ❌ FAIL: $name (HTTP:$HTTP_CODE, API:$API_CODE, expected:$expected_code)"
        FAIL=$((FAIL+1))
    fi
}

echo ""
```
