**统一响应格式**：`backend/conference-backend/conference-common/src/main/java/com/conference/common/Result.java`

```java
package com.conference.common;

import lombok.Data;
import java.io.Serializable;

@Data
public class Result<T> implements Serializable {
    private static final long serialVersionUID = 1L;
    
    private Integer code;
    private String message;
    private T data;
    private Long timestamp;
    
    public Result() {
        this.timestamp = System.currentTimeMillis();
    }
    
    public static <T> Result<T> success(T data) {
        Result<T> result = new Result<>();
        result.setCode(200);
        result.setMessage("操作成功");
        result.setData(data);
        return result;
    }
    
    public static <T> Result<T> success(String message, T data) {
        Result<T> result = new Result<>();
        result.setCode(200);
        result.setMessage(message);
        result.setData(data);
        return result;
    }
    
    public static <T> Result<T> error(String message) {
        Result<T> result = new Result<>();
        result.setCode(500);
        result.setMessage(message);
        return result;
    }
    
    public static <T> Result<T> error(Integer code, String message) {
        Result<T> result = new Result<>();
        result.setCode(code);
        result.setMessage(message);
        return result;
    }
}
```

**统一异常处理**：`backend/conference-backend/conference-common/src/main/java/com/conference/common/GlobalExceptionHandler.java`

```java
package com.conference.common;

import lombok.extern.slf4j.Slf4j;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.web.bind.annotation.RestControllerAdvice;

@Slf4j
@RestControllerAdvice
public class GlobalExceptionHandler {
    
    @ExceptionHandler(Exception.class)
    public Result<?> handleException(Exception e) {
        log.error("系统异常", e);
        return Result.error("系统异常：" + e.getMessage());
    }
    
    @ExceptionHandler(BusinessException.class)
    public Result<?> handleBusinessException(BusinessException e) {
        log.warn("业务异常：{}", e.getMessage());
        return Result.error(e.getCode(), e.getMessage());
    }
}
```

```java
package com.conference.registration.service.impl;

import com.baomidou.mybatisplus.extension.service.impl.ServiceImpl;
import com.conference.common.tenant.TenantContextHolder;
import com.conference.registration.entity.Conference;
import com.conference.registration.mapper.ConferenceMapper;
import com.conference.registration.service.ConferenceService;
import org.springframework.stereotype.Service;

@Service
public class ConferenceServiceImpl extends ServiceImpl<ConferenceMapper, Conference> 
        implements ConferenceService {
    
    @Override
    public Conference createConference(Conference conference) {
        conference.setTenantId(Long.parseLong(TenantContextHolder.getTenantId()));
        conference.setStatus("preparing");
        save(conference);
        return conference;
    }
    
    @Override
    public Conference getConferenceById(Long id) {
        return getById(id);
    }
    
    @Override
    public void updateConference(Conference conference) {
        updateById(conference);
    }
    
    @Override
    public void deleteConference(Long id) {
        removeById(id);
    }
}
```

### 3.4 会议Controller

**位置**：`backend/conference-backend/conference-registration/src/main/java/com/conference/registration/controller/ConferenceController.java`

```java
package com.conference.registration.controller;

import com.conference.common.Result;
import com.conference.registration.entity.Conference;
import com.conference.registration.service.ConferenceService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;

@RestController
@RequestMapping("/api/conference")
public class ConferenceController {
    
    @Autowired
    private ConferenceService conferenceService;
    
    @PostMapping("/create")
    public Result<Conference> createConference(@RequestBody Conference conference) {
        Conference created = conferenceService.createConference(conference);
        return Result.success(created);
    }
    
    @GetMapping("/{id}")
    public Result<Conference> getConference(@PathVariable Long id) {
        Conference conference = conferenceService.getConferenceById(id);
        return Result.success(conference);
    }
    
    @PutMapping("/{id}")
    public Result<Void> updateConference(@PathVariable Long id, 
                                        @RequestBody Conference conference) {
        conference.setId(id);
        conferenceService.updateConference(conference);
        return Result.success(null);
    }
    
    @DeleteMapping("/{id}")
    public Result<Void> deleteConference(@PathVariable Long id) {
        conferenceService.deleteConference(id);
        return Result.success(null);
    }
}
```
