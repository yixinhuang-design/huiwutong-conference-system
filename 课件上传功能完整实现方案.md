```javascript
// 上传单个附件
uploadAttachment(scheduleId, file, description = '') {
    const formData = new FormData();
    formData.append('file', file);
    if (description) {
        formData.append('description', description);
    }

    return fetch(`${this.baseURL}/attachment/upload?scheduleId=${scheduleId}`, {
        method: 'POST',
        headers: {
            'Authorization': this.getAuthToken(),
        },
        body: formData
    }).then(response => this.handleResponse(response));
}

// 批量上传附件
uploadAttachmentBatch(scheduleId, files) {
    const formData = new FormData();
    for (let file of files) {
        formData.append('files', file);
    }

    return fetch(`${this.baseURL}/attachment/upload-batch?scheduleId=${scheduleId}`, {
        method: 'POST',
        headers: {
            'Authorization': this.getAuthToken(),
        },
        body: formData
    }).then(response => this.handleResponse(response));
}

// 获取附件列表
listAttachments(scheduleId) {
    return fetch(`${this.baseURL}/attachment/list?scheduleId=${scheduleId}`, {
        headers: {
            'Authorization': this.getAuthToken(),
        }
    }).then(response => this.handleResponse(response));
}

// 删除附件
deleteAttachment(attachmentId) {
    return fetch(`${this.baseURL}/attachment/${attachmentId}`, {
        method: 'DELETE',
        headers: {
            'Authorization': this.getAuthToken(),
        }
    }).then(response => this.handleResponse(response));
}

// 获取下载地址
getDownloadUrl(attachmentId) {
    return fetch(`${this.baseURL}/attachment/${attachmentId}/download-url`, {
        headers: {
            'Authorization': this.getAuthToken(),
        }
    }).then(response => this.handleResponse(response));
}

// 记录下载
recordDownload(attachmentId) {
    return fetch(`${this.baseURL}/attachment/${attachmentId}/download`, {
        method: 'POST',
        headers: {
            'Authorization': this.getAuthToken(),
        }
    }).then(response => this.handleResponse(response));
}
```

### 3. Vue方法实现

**位置**: `admin-pc/conference-admin-pc/pages/schedule-mgr.html` (Vue methods)

```javascript
// 处理文件上传
async handleFileUpload(event) {
    const files = event.target.files;
    if (!files || files.length === 0) return;

    // 检查是否已创建日程（需要scheduleId）
    if (!this.scheduleForm.id) {
        this.$showToast('请先保存日程，再上传课件', 'warning');
        event.target.value = '';
        return;
    }

    this.$showLoading('正在上传课件...');

    try {
        const fileArray = Array.from(files);
        const uploadPromises = fileArray.map(file => 
            ScheduleAPI.uploadAttachment(this.scheduleForm.id, file, '')
                .catch(error => {
                    console.warn(`文件 ${file.name} 上传失败:`, error);
                    return null;
                })
        );

        const results = await Promise.all(uploadPromises);
        const successCount = results.filter(r => r !== null).length;

        if (!this.scheduleForm.attachments) {
            this.scheduleForm.attachments = [];
        }

        // 添加成功上传的附件到列表
        results.forEach((attachment, index) => {
            if (attachment) {
                this.scheduleForm.attachments.push({
                    id: attachment.id,
                    name: attachment.fileName,
                    size: attachment.fileSize,
                    type: attachment.fileType,
                    url: attachment.fileUrl,
                    description: attachment.description,
                    uploadTime: attachment.uploadTime
                });
            }
        });

        this.$hideLoading();
        if (successCount > 0) {
            this.$showToast(`成功上传 ${successCount} 个文件`, 'success');
        }
        if (successCount < fileArray.length) {
            this.$showToast(`有 ${fileArray.length - successCount} 个文件上传失败`, 'warning');
        }
    } catch (error) {
        this.$hideLoading();
        console.error('批量上传课件失败:', error);
        this.$showToast('上传课件失败：' + error.message, 'error');
    }

    // 清空input，允许重复上传同一文件
    event.target.value = '';
}

// 删除文件
async removeFile(index) {
    if (!this.scheduleForm.attachments || index < 0) return;

    const file = this.scheduleForm.attachments[index];
    
    // 如果文件有ID说明已保存，需要调用API删除
    if (file && file.id) {
        if (!confirm('确定要删除此课件吗？')) {
            return;
        }

        this.$showLoading('正在删除...');
        try {
            await ScheduleAPI.deleteAttachment(file.id);
            this.scheduleForm.attachments.splice(index, 1);
            this.$hideLoading();
            this.$showToast('课件已删除', 'success');
        } catch (error) {
            this.$hideLoading();
            console.error('删除课件失败:', error);
            this.$showToast('删除课件失败：' + error.message, 'error');
        }
    } else {
        // 未保存的文件直接从本地列表删除
        this.scheduleForm.attachments.splice(index, 1);
    }
}

// 下载课件
async downloadAttachment(file) {
    if (!file || !file.id) {
        this.$showToast('无法下载此文件', 'warning');
        return;
    }

    try {
        // 记录下载
        await ScheduleAPI.recordDownload(file.id);

        // 获取下载地址
        const result = await ScheduleAPI.getDownloadUrl(file.id);
        const downloadUrl = result.downloadUrl;

        // 创建隐藏的链接并触发下载
        const link = document.createElement('a');
        link.href = downloadUrl;
        link.download = file.name || 'download';
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);

        this.$showToast('开始下载：' + file.name, 'success');
    } catch (error) {
        console.error('下载课件失败:', error);
        this.$showToast('下载失败：' + error.message, 'error');
    }
}

// 编辑日程时加载附件
async editSchedule(schedule) {
    this.isEditMode = true;
    this.scheduleForm = JSON.parse(JSON.stringify(schedule));
    
    // 加载附件列表
    if (schedule.id) {
        try {
            const attachments = await ScheduleAPI.listAttachments(schedule.id);
            if (Array.isArray(attachments) && attachments.length > 0) {
                this.scheduleForm.attachments = attachments.map(att => ({
                    id: att.id,
                    name: att.fileName,
                    size: att.fileSize,
                    type: att.fileType,
                    url: att.fileUrl,
                    description: att.description,
                    uploadTime: att.uploadTime
                }));
            } else {
                this.scheduleForm.attachments = [];
            }
        } catch (error) {
            console.warn('加载附件列表失败:', error);
            this.scheduleForm.attachments = [];
        }
    }
    
    this.showModal = true;
}
```

### 3. 关键业务逻辑

#### 文件验证流程

```
文件选择
  ↓
前端验证: 文件大小、类型
  ↓ 
FormData上传
  ↓
后端控制器接收
  ↓
验证参数: scheduleId、tenantId、文件非空
  ↓
Service处理:
  ├─ 验证日程存在性
  ├─ 验证文件大小（≤100MB）
  ├─ 验证文件类型（白名单）
  ├─ 计算SHA256哈希
  ├─ 检查是否重复（防止重复上传）
  ├─ 保存文件到服务器
  ├─ 保存元数据到数据库
  └─ 返回附件信息
  ↓
前端更新UI: 显示在列表中
```

#### 文件存储路径

```
{project_root}/
└── uploads/
    └── {tenantId}/           # 按租户隔离
        └── schedule/
            └── {scheduleId}/  # 按日程分组
                └── {date}/    # 按日期分组 (yyyyMMdd)
                    └── {timestamp}_{originalFileName}
```

**例**: `/uploads/2027317834622709762/schedule/1001/20260311/1710141600000_presentation.pptx`

#### 哈希计算（防重复）

```
SHA256算法:
1. 读取整个文件的字节流
2. 使用MessageDigest.getInstance("SHA-256")计算
3. 转换为16进制字符串
4. 存储在conf_schedule_attachment.file_hash字段
5. 上传前检查是否已存在此哈希值
```

## 📊 数据库设计

### 表结构: conf_schedule_attachment

**已在V7__schedule_schema.sql中定义**

```sql
CREATE TABLE conf_schedule_attachment (
    id BIGINT PRIMARY KEY,
    schedule_id BIGINT NOT NULL,
    meeting_id BIGINT NOT NULL,
    tenant_id BIGINT NOT NULL,
    file_name VARCHAR(255) NOT NULL,
    file_size BIGINT,
    file_type VARCHAR(50),
    file_url VARCHAR(500) NOT NULL,
    file_hash VARCHAR(255),  -- SHA256哈希值，用于防重复
    description VARCHAR(255),
    upload_by BIGINT,
    upload_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    download_count INT DEFAULT 0,  -- 下载次数统计
    deleted TINYINT DEFAULT 0,
    
    KEY idx_schedule_id (schedule_id),
    KEY idx_meeting_id (meeting_id),
    KEY idx_tenant_id (tenant_id),
    UNIQUE KEY uk_schedule_file (schedule_id, file_hash, deleted)
);
```

### 字段说明

| 字段 | 类型 | 说明 | 备注 |
|------|------|------|------|
| id | BIGINT | 主键 | 唯一标识 |
| schedule_id | BIGINT | 日程ID | 关联conf_schedule |
| meeting_id | BIGINT | 会议ID | 冗余存储用于查询 |
| tenant_id | BIGINT | 租户ID | 多租户隔离 |
| file_name | VARCHAR | 原始文件名 | 如: presentation.pptx |
| file_size | BIGINT | 文件大小(字节) | 用于展示和验证 |
| file_type | VARCHAR | 文件类型 | 如: pptx, pdf, docx |
| file_url | VARCHAR | 访问URL | 如: /uploads/xxx.pptx |
| file_hash | VARCHAR | SHA256哈希 | 防止重复上传 |
| description | VARCHAR | 文件描述 | 用户备注 |
| upload_by | BIGINT | 上传人ID | 来自TenantContextHolder |
| upload_time | DATETIME | 上传时间 | 自动timestamp |
| download_count | INT | 下载次数 | 统计下载 |
| deleted | TINYINT | 删除标记 | 0=正常, 1=已删除 |

## 🔧 问题解决

### 常见问题

#### Q1: 上传失败 - "请先保存日程，再上传课件"

**原因**: 没有scheduleId（日程未保存）

**解决**:
1. 先保存日程信息
2. 获得日程ID后再上传课件
3. 检查前端代码: `if (!this.scheduleForm.id)` 的验证

#### Q2: 上传失败 - "文件大小不能超过 100MB"

**原因**: 文件超过限制

**解决**:
1. 检查文件大小
2. 如需修改限制，修改后端代码:
   ```java
   private static final long MAX_FILE_SIZE = 100 * 1024 * 1024; // 修改这里
   ```

#### Q3: 上传失败 - "不支持的文件类型"

**原因**: 文件类型不在白名单中

**解决**:
1. 检查文件扩展名
2. 如需支持新类型，修改后端代码:
   ```java
   private static final String[] ALLOWED_EXTENSIONS = {
       "ppt", "pptx", "pdf", "doc", "docx", "xls", "xlsx", 
       "txt", "jpg", "jpeg", "png"  // 在这里添加
   };
   ```

#### Q4: 文件保存位置不对

**检查步骤**:
1. 查看后端日志: `[ScheduleAttachmentServiceImpl] 文件保存成功 - fileUrl: ...`
2. 文件应该在: `{project_root}/uploads/{tenantId}/schedule/{scheduleId}/{date}/`
3. 如果目录不存在，系统会自动创建

#### Q5: 下载时提示"文件不存在"

**原因**: 文件被删除或路径不对

**解决**:
1. 检查文件是否物理存在
2. 检查路径是否正确
3. 查看数据库中file_url字段值
4. 注意路径分隔符（Windows用反斜杠需要转义）

#### Q6: 跨域错误 (CORS)

**检查**:
1. 后端Controller已添加 `@CrossOrigin(origins = "*", maxAge = 3600)`
2. 前端API已正确添加Authorization header
3. 确保baseURL正确: `http://localhost:8084/api/schedule`
