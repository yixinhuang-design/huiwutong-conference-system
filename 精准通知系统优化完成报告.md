// 批量删除
const selectedRecords = ref([]);
const selectAllRecords = ref(false);

const toggleSelectRecord = (record) => {
    const index = selectedRecords.value.indexOf(record.id);
    if (index > -1) {
        selectedRecords.value.splice(index, 1);
    } else {
        selectedRecords.value.push(record.id);
    }
};

const batchDeleteRecords = async () => {
    if (selectedRecords.value.length === 0) {
        this.$showToast('请先选择要删除的记录', 'warning');
        return;
    }
    
    const confirmed = await this.$showConfirm(
        `确定要删除选中的 ${selectedRecords.value.length} 条记录吗？`,
        '批量删除确认'
    );
    
    if (confirmed) {
        try {
            // 调用批量删除接口（如果后端有）或循环调用单条删除
            for (const id of selectedRecords.value) {
                await fetch(`/api/notification/${id}`, {
                    method: 'DELETE',
                    headers: {
                        'Authorization': `Bearer ${localStorage.getItem('authToken')}`
                    }
                });
            }
            
            // 重新加载历史记录
            await loadHistory();
            
            selectedRecords.value = [];
            selectAllRecords.value = false;
            
            this.$showToast(`成功删除 ${selectedRecords.value.length} 条记录`, 'success');
        } catch (error) {
            console.error('批量删除失败:', error);
            this.$showToast('删除失败：' + error.message, 'error');
        }
    }
};

const getFileIcon = (filename) => {
    const ext = filename.split('.').pop().toLowerCase();
    const iconMap = {
        'pdf': 'fas fa-file-pdf',
        'doc': 'fas fa-file-word',
        'docx': 'fas fa-file-word',

        'rar': 'fas fa-file-archive'
    };
    return iconMap[ext] || 'fas fa-file';

### 步骤3：实现发送逻辑

```java
@Service
public class NotificationChannelService {
    
    @Autowired
    private com.aliyun.dysmsapi20170525.Client smsClient;
    
    /**
     * 发送短信
     */
    public void sendSMS(String phoneNumber, String content) {
        com.aliyun.dysmsapi20170525.models.SendSmsRequest request = 
            new com.aliyun.dysmsapi20170525.models.SendSmsRequest()
                .setSignName("智能会议系统")
                .setTemplateCode("SMS_123456789")
                .setPhoneNumbers(phoneNumber)
                .setTemplateParam("{\"content\":\"" + content + "\"}");
        
        try {
            com.aliyun.dysmsapi20170525.models.SendSmsResponse response = 
                smsClient.sendSms(request);
            
            if (response.getBody().getCode().equals("OK")) {
                log.info("短信发送成功: {}", phoneNumber);
            } else {
                log.error("短信发送失败: {}, {}", phoneNumber, response.getBody().getMessage());
            }
        } catch (Exception e) {
            log.error("短信发送异常: {}", phoneNumber, e);
        }
    }
    
    /**
     * UniPush推送
     */
    public void sendUniPush(String clientId, String title, String content) {
        com.getui.push.SdkApi sdkApi = new com.getui.push.SdkApi(APP_ID, APP_KEY, MASTER_SECRET);
        
        com.getui.push.dto.PushDTO pushDTO = new com.getui.push.dto.PushDTO();
        pushDTO.setTitle(title);
        pushDTO.setContent(content);
        pushDTO.setPayload(new HashMap<>());
        
        try {
            com.getui.push.dto.ResultDTO result = sdkApi.push(APP_ID, pushDTO);
            if (result.getCode() == 0) {
                log.info("推送发送成功: {}", clientId);
            } else {
                log.error("推送发送失败: {}, {}", clientId, result.getMsg());
            }
        } catch (Exception e) {
            log.error("推送发送异常: {}", clientId, e);
        }
    }
}
```
