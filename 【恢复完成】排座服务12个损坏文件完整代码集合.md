package com.conference.seating.service.impl;

import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;

import org.springframework.stereotype.Service;

import jakarta.annotation.Resource;

/**
 * 会场管理服务实现

@Slf4j
@Service
public class SeatingVenueServiceImpl implements SeatingVenueService {
    
    @Resource
    private SeatingVenueMapper seatingVenueMapper;
    
    @Resource
    private SeatingSeatMapper seatingSeatMapper;
    
    @Override
    public List<VenueDetailResponse> getVenuesByConference(Long conferenceId) {
        Long tenantId = TenantContextHolder.getTenantId();
        log.info("[租户{}] 获取会议{}的会场列表", tenantId, conferenceId);
        
        LambdaQueryWrapper<SeatingVenue> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(SeatingVenue::getConferenceId, conferenceId)
               .eq(SeatingVenue::getTenantId, tenantId)
               .orderByDesc(SeatingVenue::getCreatedAt);
        
        List<SeatingVenue> venues = seatingVenueMapper.selectList(wrapper);
        
        return venues.stream()
                .map(this::buildVenueDetailResponse)
                .collect(Collectors.toList());
    }
    
    @Override
    public VenueDetailResponse getVenueDetail(Long venueId) {
        Long tenantId = TenantContextHolder.getTenantId();
        log.info("[租户{}] 获取会场{}详情", tenantId, venueId);
        
        SeatingVenue venue = getVenueByIdAndTenant(venueId, tenantId);
        return buildVenueDetailResponse(venue);
    }
    
    @Override
    @Transactional(rollbackFor = Exception.class)
    public VenueDetailResponse createVenue(VenueCreateRequest request) {
        Long tenantId = TenantContextHolder.getTenantId();
        Long userId = TenantContextHolder.getUserId();
        log.info("[租户{}] 创建会场: {}", tenantId, request.getVenueName());
        
        // 验证会议存在（可选，取决于项目架构）
        // validateConferenceExists(request.getConferenceId());
        
        SeatingVenue venue = new SeatingVenue();
        venue.setConferenceId(request.getConferenceId());
        venue.setTenantId(tenantId);
        venue.setVenueName(request.getVenueName());
        venue.setVenueType(request.getVenueType());
        venue.setTotalRows(request.getTotalRows());
        venue.setTotalColumns(request.getTotalColumns());
        venue.setCapacity(request.getCapacity());
        venue.setLayoutData(request.getLayoutData());
        venue.setStatus("ACTIVE");
        venue.setCreatedBy(userId);
        venue.setCreatedAt(LocalDateTime.now());
        
        int rows = seatingVenueMapper.insert(venue);
        if (rows <= 0) {
            throw new BusinessException(ResultCode.SYSTEM_ERROR, "会场创建失败");
        }
        
        log.info("[租户{}] 会场创建成功，ID: {}", tenantId, venue.getId());
        return buildVenueDetailResponse(venue);
    }
    
    @Override
    @Transactional(rollbackFor = Exception.class)
    public VenueDetailResponse updateVenue(Long venueId, VenueUpdateRequest request) {
        Long tenantId = TenantContextHolder.getTenantId();
        Long userId = TenantContextHolder.getUserId();
        log.info("[租户{}] 更新会场{}", tenantId, venueId);
        
        SeatingVenue venue = getVenueByIdAndTenant(venueId, tenantId);
        
        // 只更新非null字段
        if (request.getVenueName() != null) {
            venue.setVenueName(request.getVenueName());

            venue.setTotalRows(request.getTotalRows());

            venue.setTotalColumns(request.getTotalColumns());

        if (request.getCapacity() != null) {
            venue.setCapacity(request.getCapacity());
        }
        if (request.getLayoutData() != null) {
            venue.setLayoutData(request.getLayoutData());

        if (request.getStatus() != null) {
            venue.setStatus(request.getStatus());
        }
        
        venue.setUpdatedBy(userId);
        venue.setUpdatedAt(LocalDateTime.now());
        
        int rows = seatingVenueMapper.updateById(venue);
        if (rows <= 0) {
            throw new BusinessException(ResultCode.SYSTEM_ERROR, "会场更新失败");
        }
        
        log.info("[租户{}] 会场{}更新成功", tenantId, venueId);
        return buildVenueDetailResponse(venue);
    }
    
    @Override
    @Transactional(rollbackFor = Exception.class)
    public void deleteVenue(Long venueId) {
        Long tenantId = TenantContextHolder.getTenantId();
        log.info("[租户{}] 删除会场{}", tenantId, venueId);
        
        SeatingVenue venue = getVenueByIdAndTenant(venueId, tenantId);
        
        // 检查会场是否有座位分配
        LambdaQueryWrapper<SeatingSeat> seatWrapper = new LambdaQueryWrapper<>();
        seatWrapper.eq(SeatingSeat::getVenueId, venueId)
                   .eq(SeatingSeat::getTenantId, tenantId);
        long seatCount = seatingSeatMapper.selectCount(seatWrapper);
        
        if (seatCount > 0) {
            throw new BusinessException(ResultCode.INVALID_REQUEST, "会场存在座位配置，无法删除");
        }
        
        int rows = seatingVenueMapper.deleteById(venueId);
        if (rows <= 0) {
            throw new BusinessException(ResultCode.SYSTEM_ERROR, "会场删除失败");
        }
        
        log.info("[租户{}] 会场{}删除成功", tenantId, venueId);
    }
    
    @Override
    public boolean venueExists(Long venueId) {
        Long tenantId = TenantContextHolder.getTenantId();
        LambdaQueryWrapper<SeatingVenue> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(SeatingVenue::getId, venueId)
               .eq(SeatingVenue::getTenantId, tenantId);
        return seatingVenueMapper.selectCount(wrapper) > 0;
    }
    
    @Override
    public VenueSeatStatsDto getSeatStats(Long venueId) {
        Long tenantId = TenantContextHolder.getTenantId();
        log.info("[租户{}] 获取会场{}座位统计", tenantId, venueId);
        
        // 验证会场存在
        getVenueByIdAndTenant(venueId, tenantId);
        
        // 查询座位统计
        LambdaQueryWrapper<SeatingSeat> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(SeatingSeat::getVenueId, venueId)
               .eq(SeatingSeat::getTenantId, tenantId);
        
        List<SeatingSeat> seats = seatingSeatMapper.selectList(wrapper);
        
        int totalSeats = seats.size();
        int assignedSeats = (int) seats.stream().filter(s -> "assigned".equals(s.getStatus())).count();
        int availableSeats = (int) seats.stream().filter(s -> "available".equals(s.getStatus())).count();
        int reservedSeats = (int) seats.stream().filter(s -> "reserved".equals(s.getStatus())).count();
        int vipSeats = (int) seats.stream().filter(s -> "vip".equals(s.getSeatType())).count();
        int aisleSeats = (int) seats.stream().filter(s -> "aisle".equals(s.getSeatType())).count();
        
        double utilizationRate = totalSeats > 0 ? (double) assignedSeats / totalSeats * 100 : 0;
        
        return VenueSeatStatsDto.builder()
                .venueId(venueId)
                .totalSeats(totalSeats)
                .assignedSeats(assignedSeats)
                .availableSeats(availableSeats)
                .reservedSeats(reservedSeats)
                .vipSeats(vipSeats)
                .aisleSeats(aisleSeats)
                .utilizationRate(utilizationRate)
                .build();
    }

    private SeatingVenue getVenueByIdAndTenant(Long venueId, Long tenantId) {

        wrapper.eq(SeatingVenue::getId, venueId)

                .capacity(venue.getCapacity())
                .layoutData(venue.getLayoutData())
                .status(venue.getStatus())
                .createdBy(venue.getCreatedBy())
                .createdAt(venue.getCreatedAt())
                .updatedBy(venue.getUpdatedBy())
                .updatedAt(venue.getUpdatedAt())
                .seatStats(getSeatStats(venue.getId()))
                .build();
    }
