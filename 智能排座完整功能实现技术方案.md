## 二、核心功能模块实现

### 2.1 会场布局配置模块

**功能描述**：
- 支持三种会议类型：普通式、U型、圆桌
- 动态创建座位区，支持自定义行数、列数、角度
- 设置固定座位和过道位置
- 支持任意角度旋转布局

**技术实现**：
```javascript
// 座位区生成算法
generateSeatingLayout(config) {
  const { rows, cols, angle, type } = config;
  const seats = [];
  
  for (let r = 1; r <= rows; r++) {
    for (let c = 1; c <= cols; c++) {
      const seat = {
        id: `${config.areaId}${r}_${c}`,
        row: r,
        col: c,
        x: this.calculateX(r, c, angle),
        y: this.calculateY(r, c, angle),
        status: 'available',
        type: this.determineSeatType(r, c, config)
      };
      seats.push(seat);
    }
  }
  
  return seats;
}

// 角度计算
calculateX(row, col, angle) {
  const radian = (angle * Math.PI) / 180;
  return col * Math.cos(radian) - row * Math.sin(radian);
}

calculateY(row, col, angle) {
  const radian = (angle * Math.PI) / 180;
  return col * Math.sin(radian) + row * Math.cos(radian);
}
```

### 2.2 智能排座算法模块

**支持的排座算法**：

1. **智能推荐算法**（综合考虑多因素）
   - VIP优先安排（前排居中）
   - 特殊需求优先（轮椅、无障碍等）
   - 同部门相邻，职级有序
   - 性别均衡分布

2. **按部门排座算法**
   - 部门内部集中
   - 按部门人数排序分配区域

3. **按职级排座算法**
   - 职级高低决定座位位置
   - 前排优先高职级

4. **随机排座算法**
   - 完全随机分配
   - 保证VIP位置

**核心算法实现**：
```javascript
// 智能推荐算法
async executeSmartAlgorithm() {
  // 1. VIP优先安排
  const vipAttendees = this.getVipAttendees();
  await this.assignVipSeats(vipAttendees);
  
  // 2. 特殊需求安排
  const specialAttendees = this.getSpecialNeedsAttendees();
  await this.assignSpecialNeedsSeats(specialAttendees);
  
  // 3. 按部门和职级安排剩余人员
  const remaining = this.getRemainingAttendees();
  await this.assignByDepartmentAndPosition(remaining);
  
  // 4. 应用规则优化
  this.applySeatingRules();
}

// VIP座位分配（前排居中）
async assignVipSeats(vipAttendees) {
  const vipSeats = this.getFrontCenterSeats(vipAttendees.length);
  
  vipAttendees.forEach((attendee, index) => {
    if (vipSeats[index]) {
      this.assignAttendeeToSeat(attendee, vipSeats[index]);
    }
  });
}

// 获取前排居中座位
getFrontCenterSeats(count) {
  const allSeats = this.getAllAvailableSeats();
  
  return allSeats
    .sort((a, b) => {
      // 行优先（前排）
      if (a.row !== b.row) return a.row - b.row;
      
      // 同一行按居中程度排序
      const rowSeats = this.getRowSeats(a.areaId, a.row);
      const center = Math.floor(rowSeats.length / 2);
      const distA = Math.abs(a.col - center);
      const distB = Math.abs(b.col - center);
      return distA - distB;
    })
    .slice(0, count);
}
```

### 2.4 辅助安排智能分配模块

**功能描述**：
- 智能分配乘车（根据车辆容量）
- 智能分配住宿（单住/双住匹配）
- 智能分配讨论室（按小组容量）
- 自动关联参会者信息

**技术实现**：
```javascript
// 智能分配乘车
async autoAssignTransport() {
  const vehicles = this.transportList;
  const attendees = this.getAttendeesWithoutTransport();
  
  // 按部门分组分配到同一车辆
  const deptGroups = this.groupAttendeesByDepartment(attendees);
  
  for (const vehicle of vehicles) {
    const remainingCapacity = vehicle.capacity - vehicle.assignedCount;
    if (remainingCapacity <= 0) continue;
    
    // 优先分配同部门人员
    for (const dept in deptGroups) {
      if (deptGroups[dept].length === 0) continue;
      
      const assignCount = Math.min(remainingCapacity, deptGroups[dept].length);
      const assigned = deptGroups[dept].splice(0, assignCount);
      
      assigned.forEach(attendee => {
        this.assignAttendeeToVehicle(attendee, vehicle);
      });
      
      vehicle.assignedCount += assignCount;
      if (vehicle.assignedCount >= vehicle.capacity) break;
    }
  }
}

// 智能分配住宿
async autoAssignAccommodation() {
  const rooms = this.accommodationList;
  const attendees = this.getAttendeesWithoutAccommodation();
  
  // 考虑性别和部门因素
  const maleAttendees = attendees.filter(a => a.gender === '男');
  const femaleAttendees = attendees.filter(a => a.gender === '女');
  
  for (const room of rooms) {
    const remainingCapacity = room.capacity - room.assignedCount;
    if (remainingCapacity <= 0) continue;
    
    let assigned = [];
    
    if (room.type === '单人间') {
      // 单人间随机分配
      const candidate = remainingCapacity === 1 ? 
        [...maleAttendees, ...femaleAttendees][0] : null;
      if (candidate) {
        assigned = [candidate];
      }
    } else {
      // 双人间尽量同性别
      const candidates = remainingCapacity === 2 ? 
        (maleAttendees.length >= 2 ? maleAttendees : 
         femaleAttendees.length >= 2 ? femaleAttendees : 
         [maleAttendees[0], femaleAttendees[0]].filter(Boolean)) : 
        [remainingCapacity === 1 ? 
         maleAttendees[0] || femaleAttendees[0] : null].filter(Boolean);
      
      assigned = candidates.slice(0, remainingCapacity);
    }
    
    assigned.forEach(attendee => {
      this.assignAttendeeToRoom(attendee, room);
      // 从待分配列表中移除
      const index = attendees.indexOf(attendee);
      if (index > -1) attendees.splice(index, 1);
    });
    
    room.assignedCount += assigned.length;
  }
}
```

```javascript
// 查座页面逻辑（独立页面）
class SeatChecker {
  async checkSeat(name, phone) {
    // 通过姓名或手机号查询座位
    const attendee = await this.findAttendee(name, phone);
    if (!attendee) {
      throw new Error('未找到参会者信息');
    }
    
    const seat = await this.findAttendeeSeat(attendee.id);
    if (!seat) {
      throw new Error('座位信息未分配');
    }
    
    // 获取周围座位信息
    const nearbySeats = this.getNearbySeats(seat);
    
    // 获取辅助安排信息
    const arrangements = await this.getAttendeeArrangements(attendee.id);
    
    return {
      attendee,
      seat,
      nearbySeats,
      arrangements,
      // 广告信息
      advertisement: this.getRandomAd()
    };
  }
  
  getNearbySeats(seat) {
    const row = this.getRowSeats(seat.areaId, seat.row);
    const seatIndex = row.findIndex(s => s.id === seat.id);
    
    return [
      row[seatIndex - 1], // 左侧
      row[seatIndex + 1], // 右侧
      // 前后排中间位置
      this.getSeatByPosition(seat.areaId, seat.row - 1, seat.col),
      this.getSeatByPosition(seat.areaId, seat.row + 1, seat.col)
    ].filter(Boolean);
  }
}
```
