### 3. 关键集成点

#### 统计卡片更新
```javascript
async updateStatistics() {
  try {
    const meetingId = this.currentMeetingId
    
    // 获取统计数据
    this.totalCount = await ScheduleAPI.countSchedules(meetingId)
    
    // 计算总时长（从全部日程）
    const schedules = await ScheduleAPI.allSchedules(meetingId)
    this.totalDuration = schedules.reduce((sum, s) => {
      const start = new Date(s.startTime)
      const end = new Date(s.endTime)
      return sum + (end - start) / (1000 * 60) // 分钟数
    }, 0)
    
    // 获取需要签到的日程数
    const checkinSchedules = await ScheduleAPI.getNeedCheckinSchedules(meetingId)
    this.checkinCount = checkinSchedules.length
    
    // 获取需要提醒的日程数
    const reminderSchedules = await ScheduleAPI.getNeedReminderSchedules(meetingId)
    this.reminderCount = reminderSchedules.length
  } catch (error) {
    console.error('更新统计失败:', error)
  }
}
```

#### 视图切换
```javascript
// 表格视图
async showTableView() {
  this.viewMode = 'table'
  await this.loadSchedules()
}

// 日历视图
async showCalendarView() {
  this.viewMode = 'calendar'
  const schedules = await ScheduleAPI.allSchedules(this.currentMeetingId)
  this.buildCalendarData(schedules)
}

// 构建日历数据（按日期分组）
buildCalendarData(schedules) {
  const grouped = {}
  schedules.forEach(s => {
    const date = s.startTime.split(' ')[0] // yyyy-MM-dd
    if (!grouped[date]) grouped[date] = []
    grouped[date].push(s)
  })
  this.calendarData = grouped
}
```

#### 搜索和筛选
```javascript
async filterSchedules() {
  let filtered = await ScheduleAPI.allSchedules(this.currentMeetingId)
  
  // 按标题搜索
  if (this.searchText) {
    filtered = filtered.filter(s => 
      s.title.includes(this.searchText) ||
      s.host.includes(this.searchText) ||
      s.location.includes(this.searchText)
    )
  }
  
  // 按状态筛选
  if (this.selectedStatus !== 'all') {
    const statusCode = {
      'draft': 0,
      'published': 1,
      'ongoing': 2,
      'completed': 3,
      'cancelled': 4
    }[this.selectedStatus]
    filtered = filtered.filter(s => s.status === statusCode)
  }
  
  this.displaySchedules = filtered
}
```
