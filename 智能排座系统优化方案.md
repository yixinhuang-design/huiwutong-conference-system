#### 1.1 创建前端API模块

**文件路径**：`admin-pc/conference-admin-pc/js/api/notification-api.js`

```javascript
/**
 * 通知推送 API 模块
 * 对接精致通知后端接口
 */
const NotificationAPI = {
    baseURL: '/api/notification',

    /**
     * 发送座位通知
     * @param {Array} attendees - 参会人员列表（已分配座位）
     * @param {Object} conference - 会议信息
     */
    async sendSeatNotifications(attendees, conference) {
        const notifications = attendees.filter(a => a.seatAssignment).map(attendee => {
            const seat = attendee.seatAssignment;
            return {
                conferenceId: conference.id,
                templateId: null, // 使用默认模板
                title: '🪑 座位安排通知',
                content: `${attendee.name}您好，您的座位号是：${seat.areaName}${seat.rowNumber}排${seat.seatNumber}号`,
                receivers: [attendee.id || attendee.phone],
                receiverType: attendee.id ? 'user' : 'guest',
                channels: ['app', 'sms'],
                sendType: 'immediate',
                data: {
                    type: 'seat',
                    seatArea: seat.areaName,
                    seatRow: seat.rowNumber,
                    seatCol: seat.seatNumber,
                    conferenceId: conference.id,
                    scheduleId: conference.scheduleId
                }
            };
        });

        // 批量发送
        const promises = notifications.map(notif => 
            fetch(`${this.baseURL}/send`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'Authorization': this.getAuthToken()
                },
                body: JSON.stringify(notif)
            }).then(res => res.json())
        );

        return Promise.all(promises);
    },

    /**
     * 发送所有安排通知（座位+用餐+住宿+车辆）
     */
    async sendAllArrangementNotifications(data) {
        const { attendees, conference, diningList, accommodationList, transportList } = data;

        const notifications = [];

        // 座位通知
        attendees.filter(a => a.seatAssignment).forEach(attendee => {
            const seat = attendee.seatAssignment;
            notifications.push({
                conferenceId: conference.id,
                title: '🪑 座位安排',
                content: `座位：${seat.areaName}${seat.rowNumber}排${seat.seatNumber}号`,
                receivers: [attendee.id || attendee.phone],
                receiverType: attendee.id ? 'user' : 'guest',
                channels: ['app', 'sms'],
                sendType: 'immediate',
                data: { type: 'seat', ...seat }
            });
        });

        // 用餐通知
        diningList.forEach(dining => {
            dining.attendees?.forEach(attendee => {
                notifications.push({
                    conferenceId: conference.id,
                    title: '🍽️ 用餐安排',
                    content: `用餐时间：${dining.time}，地点：${dining.location}，桌号：${dining.tableNumber}`,
                    receivers: [attendee.id || attendee.phone],
                    receiverType: attendee.id ? 'user' : 'guest',
                    channels: ['app', 'sms'],
                    sendType: 'immediate',
                    data: { type: 'dining', ...dining }
                });
            });
        });

        // 住宿通知
        accommodationList.forEach(acc => {
            acc.attendees?.forEach(attendee => {
                notifications.push({
                    conferenceId: conference.id,
                    title: '🏨 住宿安排',
                    content: `房间号：${acc.roomNumber}，入住时间：${acc.checkInTime}`,
                    receivers: [attendee.id || attendee.phone],
                    receiverType: attendee.id ? 'user' : 'guest',
                    channels: ['app', 'sms'],
                    sendType: 'immediate',
                    data: { type: 'accommodation', ...acc }
                });
            });
        });

        // 车辆通知
        transportList.forEach(vehicle => {
            vehicle.passengers?.forEach(passenger => {
                notifications.push({
                    conferenceId: conference.id,
                    title: '🚗 车辆安排',
                    content: `车辆：${vehicle.name}，出发时间：${vehicle.departureTime}`,
                    receivers: [passenger.id || passenger.phone],
                    receiverType: passenger.id ? 'user' : 'guest',
                    channels: ['app', 'sms'],
                    sendType: 'immediate',
                    data: { type: 'transport', ...vehicle }
                });
            });
        });

        // 批量发送
        const promises = notifications.map(notif => 
            fetch(`${this.baseURL}/send`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'Authorization': this.getAuthToken()
                },
                body: JSON.stringify(notif)
            }).then(res => res.json())
        );

        return Promise.all(promises);
    },

    getAuthToken() {
        const token = localStorage.getItem('authToken') || '';
        return token ? `Bearer ${token}` : '';
    }
};
```

## 2️⃣ 优化LocalStorage依赖

### 当前问题
- 座位布局、人员列表、操作历史全部存在LocalStorage
- 刷新浏览器可能丢失数据
- 无法跨设备同步

### 解决方案

#### 2.1 数据存储优先级

**优先级策略**：
```
1. 后端存储（DB） - 主要存储
2. LocalStorage - 缓存和离线fallback
3. Memory - 临时状态
```

#### 2.2 修改保存逻辑

**原代码**：
```javascript
const saveData = () => {
    const data = {
        conferenceId: conference.value?.id,
        scheduleId: currentScheduleId.value,
        areas: areas.value,
        attendees: attendees.value,
        // ...
    };
    localStorage.setItem(`seatingLayout_${key}`, JSON.stringify(data));
};
```

**修改为**：
```javascript
const saveData = async () => {
    const data = {
        conferenceId: conference.value?.id,
        scheduleId: currentScheduleId.value,
        areas: areas.value,
        attendees: attendees.value,
        // ...
    };

    try {
        // 优先保存到后端
        await SeatingAPI.saveLayout(data);
        
        // 同时保存到LocalStorage作为缓存
        localStorage.setItem(`seatingLayout_${key}`, JSON.stringify(data));
        
        console.log('数据已保存到后端和本地缓存');
    } catch (error) {
        console.warn('保存到后端失败，保存到本地:', error);
        // 后端失败时仍保存到本地
        localStorage.setItem(`seatingLayout_${key}`, JSON.stringify(data));
    }
};
```

#### 2.3 修改加载逻辑

**原代码**：
```javascript
const loadData = () => {
    const key = `${conferenceId}_${scheduleId}`;
    const data = localStorage.getItem(`seatingLayout_${key}`);
    if (data) {
        const parsed = JSON.parse(data);

```

**修改为**：
```javascript
const loadData = async () => {
    try {
        // 优先从后端加载
        const data = await SeatingAPI.loadLayout(conferenceId, scheduleId);
        
        areas.value = data.areas || [];
        attendees.value = data.attendees || [];
        
        // 缓存到本地
        localStorage.setItem(`seatingLayout_${key}`, JSON.stringify(data));
        
        console.log('已从后端加载数据');
    } catch (error) {
        console.warn('从后端加载失败，尝试从本地缓存加载:', error);
        
        // 后端失败时从本地缓存加载
        const key = `${conferenceId}_${scheduleId}`;
        const cached = localStorage.getItem(`seatingLayout_${key}`);
        if (cached) {
            const parsed = JSON.parse(cached);
            areas.value = parsed.areas || [];
            attendees.value = parsed.attendees || [];
            console.log('已从本地缓存加载数据');
        } else {
            console.warn('本地缓存也无效，使用空数据');
            areas.value = [];
            attendees.value = [];
        }
    }
};
```

#### 2.4 创建排座API模块

**文件路径**：`admin-pc/conference-admin-pc/js/api/seating-api.js`

```javascript
/**
 * 排座管理 API 模块
 */
const SeatingAPI = {
    baseURL: '/api/seating',

    /**
     * 保存座位布局
     */
    async saveLayout(data) {
        return fetch(`${this.baseURL}/layout/save`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': this.getAuthToken()
            },
            body: JSON.stringify(data)
        }).then(res => res.json());
    },

    /**
     * 加载座位布局
     */
    async loadLayout(conferenceId, scheduleId) {
        const params = new URLSearchParams({
            conferenceId: conferenceId,
            scheduleId: scheduleId || ''
        });
        return fetch(`${this.baseURL}/layout/load?${params}`, {
            headers: {
                'Authorization': this.getAuthToken()
            }
        }).then(res => res.json())
          .then(result => result.data);
    },

    /**
     * 分配座位
     */
    async assignSeat(seatId, attendeeId) {
        return fetch(`${this.baseURL}/seats/assign`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': this.getAuthToken()
            },
            body: JSON.stringify({
                seatId: seatId,
                attendeeId: attendeeId
            })
        }).then(res => res.json());
    },

    /**
     * 交换座位
     */
    async swapSeats(seatId1, seatId2) {
        return fetch(`${this.baseURL}/seats/swap`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': this.getAuthToken()
            },
            body: JSON.stringify({
                seatId1: seatId1,
                seatId2: seatId2
            })
        }).then(res => res.json());
    },

    getAuthToken() {
        const token = localStorage.getItem('authToken') || '';
        return token ? `Bearer ${token}` : '';
    }
};
```
