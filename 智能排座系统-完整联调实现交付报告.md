const loadSeatingData = async () => {
    // 首先尝试从API加载
    if (conference.value && conference.value.id) {
        try {
            // 1. 获取会议的所有会场列表
            const venuesResponse = await fetch(
              `/api/seating/venues/${conference.value.id}`,
              { method: 'GET', headers: { 'Content-Type': 'application/json' } }
            );
            
            if (venuesResponse.ok) {
                const venuesData = await venuesResponse.json();
                const venues = venuesData.data || [];
                
                // 2. 对每个会场获取座位平面图
                for (const venue of venues) {
                    const layoutResponse = await fetch(
                      `/api/seating/layout?venueId=${venue.id}`,
                      { method: 'GET', headers: { 'Content-Type': 'application/json' } }
                    );
                    
                    if (layoutResponse.ok) {
                        const layoutData = await layoutResponse.json();
                        const layout = layoutData.data || {};
                        
                        // 3. 转换API数据为前端格式
                        const seatingArea = {
                            id: `area_${venue.id}`,
                            venueId: venue.id,
                            name: venue.venueName,
                            type: venue.venueType || 'normal',
                            rows: layout.totalRows || 10,
                            cols: layout.totalColumns || 15,
                            seats: layout.seats || [],
                            stats: layout.stats || {}
                        };
                        seatingAreas.value.push(seatingArea);
                    }
                }
            }
        } catch (e) {
            console.warn('API加载失败，使用localStorage备用方案:', e);
            loadSeatingDataFromLocalStorage();
        }
    }
    
    // 参会人员初始化
    if (attendees.value.length === 0) {
        initializeDefaultAttendees();
    }
}

### 3.2 seat-manage.html 修改

**修改范围**: 座位网格初始化

**原有代码** (第110-120行)
```html
<div class="seat-grid">
    <!-- 座位图由后端渲染 -->
    <div class="seat-item occupied" title="张三">张三</div>
    <div class="seat-item vacant">空</div>
    <!-- ... 7个硬编码座位 ... -->
</div>
```

**修改为**
```html
<div class="seat-grid" id="seatGridContainer">
    <!-- 座位图由JavaScript动态生成 -->
</div>

<script>
// 动态加载座位数据
async function loadSeatsFromAPI() {
    try {
        const venueId = new URLSearchParams(window.location.search)
            .get('venueId') || localStorage.getItem('currentVenueId');
        
        if (!venueId) {
            loadDefaultSeats();
            return;
        }
        
        const response = await fetch(`/api/seating/layout?venueId=${venueId}`);
        if (response.ok) {
            const result = await response.json();
            renderSeats(result.data.seats, result.data.totalColumns);
        } else {
            loadDefaultSeats();
        }
    } catch (error) {
        console.error('加载失败:', error);
        loadDefaultSeats();
    }
}

// 渲染座位网格
function renderSeats(seats, totalColumns) {
    const container = document.getElementById('seatGridContainer');
    container.innerHTML = '';
    container.style.gridTemplateColumns = `repeat(${totalColumns || 15}, 1fr)`;
    
    // 根据行号和列号生成座位DOM
    seats.forEach(seat => {
        const seatEl = document.createElement('div');
        seatEl.className = 'seat-item';
        
        if (seat.status === 'occupied') {
            seatEl.classList.add('occupied');
            seatEl.textContent = seat.assignedUserName?.substring(0, 2);
        } else if (seat.seatType === 'vip') {
            seatEl.classList.add('vip');
            seatEl.textContent = 'VIP';
        } else {
            seatEl.classList.add('vacant');
            seatEl.textContent = '空';
        }
        
        container.appendChild(seatEl);
    });
}

// 页面加载时初始化
document.addEventListener('DOMContentLoaded', loadSeatsFromAPI);
</script>
```

**修改特点**
- ✅ 完全动态生成座位DOM
- ✅ 根据实际列数自动调整Grid布局
- ✅ 显示真实的参会人员名字
- ✅ 智能识别座位类型（普通/VIP/过道）
- ✅ 提供默认数据备用方案
