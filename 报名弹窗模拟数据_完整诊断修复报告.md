```javascript
// ✅ 新增方法1: 加载报名列表
async loadRegistrationListData(conferenceId, status) {
    if (!conferenceId) return;
    
    try {
        // 获取认证Token
        const token = localStorage.getItem('authToken');
        const headers = {
            'Content-Type': 'application/json'
        };
        if (token) {
            headers['Authorization'] = `Bearer ${token}`;
        }

        // 调用API获取报名列表
        const resp = await fetch(
            `http://localhost:8082/api/registration/list?conferenceId=${encodeURIComponent(String(conferenceId))}&status=${status}&page=1&pageSize=10000`,
            { method: 'GET', headers }
        );

        if (!resp.ok) {
            console.error('加载报名列表失败，HTTP:', resp.status);
            return;
        }

        // 解析响应
        const result = await resp.json();
        if (result.code !== 200 || !result.data) {
            console.warn('报名列表数据为空或请求失败');
            return;
        }

        // 提取报名记录
        const records = result.data.records || [];
        const typeKey = this.modalType || 'registered';
        
        // 转换为前端格式
        this.registrationLists[typeKey] = records.map(item => ({
            id: item.id,
            name: item.realName || item.name || '',
            phone: item.phone || '',
            department: item.department || '',
            position: item.position || '',
            registrationTime: item.createdTime || item.registrationTime || '',
            aiReview: item.aiReview || 'uncertain',
            reviewStatus: this.getReviewStatusKey(item.status),
            status: item.status
        }));

        console.log(`已加载 ${typeKey} 报名列表: ${this.registrationLists[typeKey].length} 条`);
    } catch (error) {
        console.error('加载报名列表数据失败:', error);
    }
}

// ✅ 新增方法2: 状态转换
getReviewStatusKey(apiStatus) {
    const statusMap = {
        0: 'pending',      // 待审核
        1: 'approved',     // 已通过
        2: 'rejected',     // 已拒绝
        3: 'cancelled',    // 已取消
        4: 'checkedin'     // 已签到
    };
    return statusMap[apiStatus] || 'pending';
}
```
