## 🛠️ 常用命令速查

### 查询租户
```sql
-- 查看所有租户
SELECT id, tenant_code, tenant_name, status FROM sys_tenant;

-- 查看活跃租户
SELECT * FROM sys_tenant WHERE status = 'active';

-- 查看租户详情（带用户数）
SELECT t.id, t.tenant_code, t.tenant_name, COUNT(u.id) as user_count
FROM sys_tenant t
LEFT JOIN sys_user u ON t.id = u.tenant_id
GROUP BY t.id;
```

### 查询用户
```sql
-- 查看租户1的所有用户
SELECT id, username, real_name, email, status FROM sys_user WHERE tenant_id = 1;

-- 查看某个用户的角色
SELECT u.username, r.role_name, t.tenant_code
FROM sys_user u
JOIN sys_user_role ur ON u.id = ur.user_id
JOIN sys_role r ON ur.role_id = r.id
JOIN sys_tenant t ON u.tenant_id = t.id
WHERE u.username = 'admin' AND u.tenant_id = 1;

-- 查看某用户的权限
SELECT DISTINCT p.permission_name, p.permission_code
FROM sys_user u
JOIN sys_user_role ur ON u.id = ur.user_id
JOIN sys_role_permission rp ON ur.role_id = rp.role_id
JOIN sys_permission p ON rp.permission_id = p.id
WHERE u.username = 'admin' AND u.tenant_id = 1;
```

### 查询角色
```sql
-- 查看租户1的所有角色
SELECT id, role_name, role_code, status FROM sys_role WHERE tenant_id = 1;

-- 查看某角色的权限
SELECT r.role_name, p.permission_name
FROM sys_role r
LEFT JOIN sys_role_permission rp ON r.id = rp.role_id
LEFT JOIN sys_permission p ON rp.permission_id = p.id
WHERE r.tenant_id = 1 AND r.role_code = 'admin';
```

### 管理用户
```sql
-- 添加新用户
INSERT INTO sys_user (id, tenant_id, username, password, email, real_name, status)
VALUES (100, 1, 'newuser', 'encrypted_pwd', 'new@email.com', '新用户', 'active');

-- 激活用户
UPDATE sys_user SET status = 'active' WHERE id = 100;

-- 禁用用户
UPDATE sys_user SET status = 'locked' WHERE id = 100;

-- 删除用户（软删除）
UPDATE sys_user SET deleted_at = NOW() WHERE id = 100;

-- 给用户分配角色
INSERT INTO sys_user_role (id, tenant_id, user_id, role_id)
VALUES (200, 1, 100, 2);  -- 将用户100分配到角色2

-- 移除用户角色
DELETE FROM sys_user_role WHERE user_id = 100 AND tenant_id = 1;
```

## 📈 数据库监控

### 存储空间
```sql
-- 查看表大小
SELECT 
  table_name,
  ROUND(((data_length + index_length) / 1024 / 1024), 2) AS size_mb
FROM information_schema.tables
WHERE table_schema = 'conference_db'
ORDER BY size_mb DESC;

-- 查看数据库总大小
SELECT 
  ROUND(SUM(data_length + index_length) / 1024 / 1024, 2) AS total_size_mb
FROM information_schema.tables
WHERE table_schema = 'conference_db';
```

### 记录数统计
```sql
-- 统计各表记录数
SELECT 
  'sys_tenant' as table_name, COUNT(*) as record_count FROM sys_tenant
UNION ALL
SELECT 'sys_user', COUNT(*) FROM sys_user
UNION ALL
SELECT 'sys_role', COUNT(*) FROM sys_role
UNION ALL
SELECT 'sys_permission', COUNT(*) FROM sys_permission
UNION ALL
SELECT 'sys_user_role', COUNT(*) FROM sys_user_role
UNION ALL
SELECT 'sys_role_permission', COUNT(*) FROM sys_role_permission
ORDER BY table_name;
```

## 🔐 多租户数据隔离验证

### 验证租户隔离
```sql
-- 租户1的用户不应该能看到租户2的用户
SELECT COUNT(*) FROM sys_user WHERE tenant_id = 1;  -- 应该得到 7 条
SELECT COUNT(*) FROM sys_user WHERE tenant_id = 2;  -- 应该得到 3 条

-- 验证用户名唯一性（仅在同一租户）
SELECT username, COUNT(*) as cnt FROM sys_user GROUP BY tenant_id, username HAVING cnt > 1;
-- 结果应为空

-- 验证租户内用户名唯一性
SELECT * FROM sys_user u1 WHERE EXISTS (
  SELECT 1 FROM sys_user u2 
  WHERE u1.tenant_id = u2.tenant_id 
  AND u1.username = u2.username 
  AND u1.id < u2.id
);
-- 结果应为空（无重复）
```

## 🆘 常见问题

### Q: 忘记了管理员密码怎么办？
```sql
-- 重置为默认密码 123456（已加密）
UPDATE sys_user 
SET password = '$2a$10$nzZg6ZbJI8JD1MQ5I5y82.qH7cq3C7Og1YJqDe5l8KPgNJlYRz9jO'
WHERE username = 'admin' AND tenant_id = 1;
```

### Q: 如何删除某个租户及其所有数据？
```sql
-- 警告：此操作不可撤销！

-- Step 1: 删除级联数据
DELETE FROM sys_user_role WHERE tenant_id = 1;
DELETE FROM sys_role_permission WHERE tenant_id = 1;
DELETE FROM sys_permission WHERE tenant_id = 1;
DELETE FROM sys_role WHERE tenant_id = 1;
DELETE FROM sys_user WHERE tenant_id = 1;

-- Step 2: 删除租户
DELETE FROM sys_tenant WHERE id = 1;
```

### Q: 如何统计某个租户有多少个用户？
```sql
SELECT COUNT(*) as user_count FROM sys_user WHERE tenant_id = 1;
```

### Q: 如何批量给用户分配角色？
```sql
-- 给租户1的所有普通用户分配"审核员"角色
INSERT INTO sys_user_role (id, tenant_id, user_id, role_id)
SELECT 
  AUTO_INCREMENT + row_num, 
  1, 
  u.id, 
  5  -- role_id 5 是审核员
FROM sys_user u
WHERE u.tenant_id = 1 
  AND u.id NOT IN (
    SELECT user_id FROM sys_user_role 
    WHERE tenant_id = 1 AND role_id = 5
  );
```
