// 按部门分组
Map<String, List<Registration>> groupedMap = registrations.stream()

// 按职位分组
Map<String, List<Registration>> groupedMap = registrations.stream()

    // 仅更新前端状态
    const index = groups.value.findIndex(g => g.id === currentGroup.value.id);

**优化后**：
```javascript
const saveGroupDetail = async () => {
    try {
        // 调用后端API保存
        if (currentGroup.value.id) {
            // 更新现有分组
            await GroupAPI.updateGroup(currentGroup.value.id, {
                name: currentGroup.value.name,
                description: currentGroup.value.description,
                color: currentGroup.value.color,
                icon: currentGroup.value.icon,
                sort: currentGroup.value.sort
            });
        } else {
            // 创建新分组
            const result = await GroupAPI.createGroup({
                conferenceId: conference.value.id,
                name: currentGroup.value.name,
                description: currentGroup.value.description,
                color: currentGroup.value.color,
                icon: currentGroup.value.icon,
                type: 'manual',
                sort: currentGroup.value.sort
            });
            currentGroup.value.id = result.id;
        }

        // 更新前端状态
        const index = groups.value.findIndex(g => g.id === currentGroup.value.id);
        if (index > -1) {
            groups.value[index] = { ...currentGroup.value };
        } else {
            groups.value.push({ ...currentGroup.value });
        }

        showGroupDetailModal.value = false;
        this.$showToast('分组保存成功', 'success');
    } catch (error) {
        console.error('保存分组失败:', error);
        this.$showToast('保存失败：' + error.message, 'error');
    }
};
```

## 📝 数据库表结构

### 表1：registration_group（分组表）
```sql
CREATE TABLE registration_group (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    conference_id BIGINT NOT NULL,
    name VARCHAR(100) NOT NULL,
    description VARCHAR(500),
    color VARCHAR(20),
    icon VARCHAR(50),
    type VARCHAR(20) DEFAULT 'manual',
    sort INT DEFAULT 0,
    create_time DATETIME,
    update_time DATETIME,
    creator_id BIGINT,
    deleted BOOLEAN DEFAULT FALSE
);
```

### 表2：registration_group_member（分组成员表）
```sql
CREATE TABLE registration_group_member (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    group_id BIGINT NOT NULL,
    registration_id BIGINT NOT NULL,
    is_staff BOOLEAN DEFAULT FALSE,
    create_time DATETIME,
    UNIQUE KEY uk_group_member (group_id, registration_id)
);
```

### 表3：registration_audit_log（审核日志表）
```sql
CREATE TABLE registration_audit_log (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    registration_id BIGINT NOT NULL,
    auditor_id BIGINT,
    auditor_name VARCHAR(50),
    action VARCHAR(20),
    comment VARCHAR(500),
    create_time DATETIME
);
```

### 表4：registration_audit_rule（审核规则表）
```sql
CREATE TABLE registration_audit_rule (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    conference_id BIGINT NOT NULL,
    rule_type VARCHAR(20),
    rule_content TEXT,
    rule_name VARCHAR(100),
    enabled BOOLEAN DEFAULT TRUE,
    priority INT DEFAULT 0
);
```
